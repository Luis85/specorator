Key algorithms

Pseudocode for the six core algorithms: gate evaluation, OQ scanning, snapshot/undo, stale detection, conflict detection, and the feature context resolver. Implementation-ready — these define exact behaviour, not just intent.

1Gate evaluation

Runs on every file save (if settings.gateScanTrigger = "on-save") or manually. Returns a GateResult — a list of check outcomes. Never mutates state.

function evaluateGate(feature: Feature, gate: "define-build" | "build-ship"): GateResult {

  const checks: CheckResult[] = []

  if (gate === "define-build") {

    // A1: Vision length
    const vision = readStepBody(feature, 1)
    checks.push({
      id: "A1",
      label: "Vision is one paragraph",
      pass: wordCount(firstParagraph(vision)) <= 150,
      severity: "hard"
    })

    // A2: EARS rule count
    const rules = readStepBody(feature, 3)
    const earsLines = rules.split("\n")
      .filter(l => /^(WHEN|IF|WHILE|WHERE)\s/.test(l.trim()))
    checks.push({
      id: "A2",
      label: `${earsLines.length} EARS rules (min ${settings.minEarsRules})`,
      pass: earsLines.length >= settings.minEarsRules,
      severity: "hard"
    })

    // A3: Zero open questions (all step files)
    const openOQs = scanOQs(feature).filter(oq => !oq.resolved)
    checks.push({
      id: "A3",
      label: `${openOQs.length} open questions`,
      pass: openOQs.length === 0,
      severity: "hard"
    })

    // A4: Decision note linked (configurable severity)
    const howItWorks = readStepBody(feature, 4)
    checks.push({
      id: "A4",
      label: "Decision note linked from step 4",
      pass: /\[\[decisions\//.test(howItWorks),
      severity: settings.decisionNoteGateCheck  // "hard"|"soft"|"off"
    })

    // A5: Constitution respected
    const rules_const = parseConstitution()
    checks.push({
      id: "A5",
      label: "House rules respected",
      pass: rules_const.every(r => isAcknowledged(r, feature)),
      severity: "hard"
    })

    // Team mode: peer sign-off
    if (settings.gateMode === "team") {
      checks.push({
        id: "T1",
        label: "Peer sign-off",
        pass: hasPeerSignoff(feature, "define-build"),
        severity: "hard"
      })
    }

  } else if (gate === "build-ship") {

    // B1: All build steps done or skipped
    const buildSteps = [5, 6, 7, 8]
    const allDone = buildSteps.every(n => {
      const s = getStepStatus(feature, n)
      return s.status === "done" || (s.status === "skipped" && s.skip_reason)
    })
    checks.push({ id: "B1", label: "All build steps done or skipped", pass: allDone, severity: "hard" })

    // B2: Constitution (re-run — may have changed)
    const rules_b = parseConstitution()
    checks.push({
      id: "B2", label: "House rules respected",
      pass: rules_b.every(r => isAcknowledged(r, feature)), severity: "hard"
    })

    // B3: Zero OQs (again)
    const openOQs_b = scanOQs(feature).filter(oq => !oq.resolved)
    checks.push({ id: "B3", label: "0 open questions", pass: openOQs_b.length === 0, severity: "hard" })

  }

  return {
    gate,
    checks,
    pass: checks.filter(c => c.severity === "hard").every(c => c.pass),
    failCount: checks.filter(c => !c.pass && c.severity !== "off").length
  }
}

2Open question scanner

Scans all step files for ? -prefixed lines. Resolution is detected by a following ! line. Runs on every file save for files in the feature folder.

function scanOQs(feature: Feature): OQRecord[] {
  const results: OQRecord[] = []

  for (const stepFile of feature.stepFiles) {
    const lines = readFile(stepFile).split("\n")

    for (let i = 0; i < lines.length; i++) {
      const line = lines[i].trim()

      if (/^\? .+/.test(line)) {
        // Found an open question line
        const questionText = line.slice(2)  // strip "? "

        // Check if next non-empty line is a resolution (! prefix)
        const nextLine = lines.slice(i + 1).find(l => l.trim() !== "")?.trim()
        const resolved = nextLine?.startsWith("! ") ?? false

        // Extract resolver handle from comment if present
        // e.g. 
        const resolverMatch = nextLine?.match(/<!--\s*@(\S+)/)
        const resolvedBy = resolverMatch ? `@${resolverMatch[1]}` : null

        results.push({
          file: stepFile,
          line: i,
          text: questionText,
          resolved,
          resolvedBy,
          age: daysSince(fileMtime(stepFile))
        })
      }
    }
  }

  return results
}

// OQ staleness check (run on cache build + on save)
function isOQStale(oq: OQRecord): boolean {
  return !oq.resolved && oq.age > settings.oqStaleThresholdDays
}

3Snapshot & undo

function snapshot(feature: Feature, label: string): SnapshotRecord {
  const snapId = `snap-${Date.now()}`
  const snapPath = `.specorator/snapshots/${feature.slug}/${snapId}/`

  // Copy all step files + _meta.md to the snapshot dir
  for (const file of feature.allFiles) {
    const content = await vault.read(file)
    await vault.create(`${snapPath}${file.name}`, content)
  }

  // Record in _meta.md snapshots[]
  const record = { id: snapId, label, timestamp: now(), path: snapPath }
  await appendToMeta(feature, "snapshots", record)
  return record
}

function undo(feature: Feature): void {
  // Find the most recent snapshot within the 24h window
  const recent = feature.meta.snapshots
    .filter(s => hoursSince(s.timestamp) < 24)
    .sort((a, b) => b.timestamp - a.timestamp)[0]

  if (!recent) throw "No undo snapshot available"

  // Restore all files from snapshot
  for (const snapFile of listFiles(recent.path)) {
    const content = await vault.read(snapFile)
    await vault.modify(featureFile(feature, snapFile.name), content)
  }

  // Remove the snapshot we just restored from (it's consumed)
  await removeSnapshot(feature, recent.id)
}

// Prune stale snapshots on plugin load
function pruneSnapshots(feature: Feature): void {
  for (const snap of feature.meta.snapshots) {
    const isGateCrossing = snap.label.startsWith("before-")
    const isExpired = hoursSince(snap.timestamp) > 24
    // Keep gate-crossing snapshots until feature is archived
    if (isExpired && !isGateCrossing) deleteSnapshot(feature, snap.id)
  }
}

4Feature context resolver

Called on every active-leaf-change event. Determines which feature the cockpit should show. Implements DR-002.

function resolveActiveFeature(activeFile: TFile | null): Feature | null {

  if (!activeFile) return lastFocusedFeature  // OQ-2: restore context

  // Check if the active file is inside a feature folder
  const path = activeFile.path
  const match = path.match(/^features\/([^/]+)\//)

  if (match) {
    const slug = match[1]
    const feature = cache.features.get(slug)
    if (feature) {
      lastFocusedFeature = feature  // persist for next session
      return feature
    }
  }

  // File is not in a feature folder — keep showing last focused
  return lastFocusedFeature
}

// Empty state logic
function emptyCockpitState(): "new-vault" | "all-archived" | "feature" {
  const active = cache.features.filter(f => f.status === "active")
  if (cache.features.size === 0) return "new-vault"
  if (active.length === 0) return "all-archived"
  return "feature"
}

5Stale detection

function isFeatureStale(feature: Feature): boolean {
  return (
    feature.status === "active" &&
    daysSince(feature.meta.last_active_at) > settings.staleThresholdDays
  )
}

// Run on cache build and on plugin load
function computeStaleFeatures(): Feature[] {
  return [...cache.features.values()].filter(isFeatureStale)
}

6Sync conflict detection

Applies in team mode only. Obsidian Sync creates duplicate files with "(conflicted copy)" suffix. Git creates files with conflict markers. Both need detection and surface — but the plugin does not resolve them automatically.

function detectConflicts(feature: Feature): ConflictRecord[] {
  const conflicts: ConflictRecord[] = []

  for (const file of feature.allFiles) {
    // Obsidian Sync: duplicate file with "(conflicted copy)" in name
    const conflictedCopy = vault.getAbstractFileByPath(
      file.path.replace(".md", " (conflicted copy).md")
    )
    if (conflictedCopy) {
      conflicts.push({ type: "sync-duplicate", file, conflictFile: conflictedCopy })
    }

    // Git: conflict markers in file content
    const content = await vault.read(file)
    if (content.includes("<<<<<<< HEAD")) {
      conflicts.push({ type: "git-markers", file, conflictFile: null })
    }
  }

  return conflicts
}

Do not auto-merge. The plugin surfaces the conflict and offers: "keep yours", "keep theirs" (for Sync duplicates), or "open in editor" (for Git markers). It never merges spec content automatically — spec text is a human decision.
